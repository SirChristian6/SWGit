	<head>

		<title>Tarea1</title>

	<body>
        
		<form>
            
			Operando1:<br>
            
			<input type="number" id="op1" name="op1"><br><br>
            
			Operador:
            
			<select id="opr" name="operator" size="1">
                
				<option value="sum">+</option>
                
				<option value="res">-</option>
                
				<option value="mul">*</option>
                
				<option value="div">/</option>
            
			</select><br><br>
             
			Operando2:<br>
            
			<input type="number" id="op2" name="op2"><br><br>
           
            
			Resultado:<br>
            
			<input type="number" id="res" name="res"><br><br>
            
			<button type="button" onClick="comprobar()">Comprobar</button>
            
			<button type="reset" onClick="borrar()">Reset</button>
            
			<div  id="correccion">
            
			</div>
        
		</form>
        
		<script>
            
			var correccion=document.getElementById("correccion");
            
			function comprobar() {
                
				var op1= parseInt(document.getElementById("op1").value);
                
				var op2= parseInt(document.getElementById("op2").value);
                
				var opr= document.getElementById("opr").value;
                
				var res= parseInt(document.getElementById("res").value);
                
				var resultado;
                
				if(opr=="sum")
                    
					resultado=op1+op2;
                
				else if(opr=="res")
                    
					resultado=op1-op2;
                
				else if(opr=="mul")
                    
					resultado=op1*op2;
                
				else if(opr=="div")
                    
					resultado=op1/op2;
                
                
				var text;  
                
				if(resultado==res)
                    
					text="Correcto";
                
				else
                    
					text="Incorrecto";
                
				correccion.innerHTML=text;
            
			}
            
			function borrar(){
                
				correccion.innerHTML="";
            
			}
        
		</script>
    
	</body>
